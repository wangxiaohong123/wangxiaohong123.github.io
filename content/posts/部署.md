```shell
# 脚本自启
# 修改sh脚本中的~，systemd不认识~符号

# 创建service文件
vim /etc/systemd/system/stars-content.service
# 插入以下内容
[Unit]
Description=Stars Content Production Service (Root)
After=network.target
[Service]
Type=forking
User=root
Group=root
# 显式使用 sh 执行，避免脚本首行解释器识别错误
ExecStart=/usr/bin/sh /root/server/content-prod.sh start
ExecStop=/usr/bin/sh /root/server/content-prod.sh stop
# 告诉 systemd，只要这个脚本成功返回 0 或 143，就认为启动成功
SuccessExitStatus=143
Restart=on-failure
RestartSec=10s
[Install]
WantedBy=multi-user.target


# 重新加载配置
sudo systemctl daemon-reload
# 设置开机自启
systemctl enable stars-content
# 手动启动服务
systemctl start stars-content
#查看服务详情
systemctl status stars-content
```
