```shell
wget https://openresty.org/download/openresty-1.19.3.1.tar.gz
# 解压、进入绕目录中
tar -zxvf openresty-1.19.3.1.tar.gz
cd openresty-1.19.3.1
# 检查环境，生成编译文件
./configure
# 出现错误ERROR: failed to run command: make TARGET_STRIP=@: CCDEBUG=-g CC=cc PREFIX=/usr/local/openresty/luajit
yum -y install gcc
# 出现错误./configure: error: the HTTP rewrite module requires the PCRE library.安装pcre-devel与openssl-devel
yum -y install pcre-devel openssl openssl-devel
# 安装完后重新执行./configure通过，进行编译
gmake && gmake install
# 编译之后在解压目录下出现openresty文件夹，可以删掉之前的解压文件
rm -rf openresty-1.19.3.1
```
